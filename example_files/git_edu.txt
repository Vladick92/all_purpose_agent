git branch - list branches
git branch -m 'name' - rename or create new branch
git branch -d 'name' - safe branch delete
git branch -D 'name' - force delete
after command above
git push origin --delete 'name'

git switch 'name' - switch to another branch
git switch -c 'name' - switch to new branch
